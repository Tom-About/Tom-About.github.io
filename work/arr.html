# 数组方法

# map

map() 这个方法返回一个新数组，数组中的元素为原数组元素调用函数处理后的值。

var users = [
  {name: "张三", "email": "zhang@email.com"},
  {name: "李四",   "email": "li@email.com"},
  {name: "马五",  "email": "ma@email.com"}
];

var emails = users.map(function (user) { return user.email; });
console.log(emails);
//["zhang@email.com", "jiang@email.com", "li@email.com"]

# filter

filter() 这个方法创建一个新的数组，新数组中的元素是通过检查指定数组中符合条件的所有元素。

var ages = [32, 33, 16, 40];

function chAdu(age) {
    return age >= 18;
}

function myFunction() {
    document.getElementById("demo").innerHTML = ages.filter(chAdu);
}
// 输出结果为：32,33,40

# every

every() 方法用于检测数组所有元素是否都符合指定条件（通过函数提供）

var ages = [32, 33, 16, 40];

function chAdu(age) {
    return age >= 18;
}

function myFunction() {
    document.getElementById("demo").innerHTML = ages.every(chAdu);
}
// 输出结果为：false

# reduce

reduce() 方法接收一个函数作为累加器，数组中的每个值（从左到右）开始缩减，最终计算为一个值。

var arr = [65, 44, 12, 4];
 
function getSum(to, num) {
    return to + num;
}
function myFunction(item) {
    document.getElementById("app").innerHTML = arr.reduce(getSum);
}
// 输出结果：125

# some

some() 方法用于检测数组中的元素是否满足指定条件（满足函数中的条件）。

var arr = [3, 10, 18, 20];

function chAdu(age) {
    return age >= 18;
}

function myFunction() {
    document.getElementById("demo").innerHTML = arr.some(chAdu);
}
// 输出结果为：true

# find

find() 方法返回通过测试（函数内进行判断）的数组的第一个元素的值。

var arr = [3, 10, 18, 20];
 
function chAdu(age) {
    return age >= 10;
}
 
function myFunction() {
    document.getElementById("demo").innerHTML = arr.find(chAdu);
}
// fruits 输出结果：10
